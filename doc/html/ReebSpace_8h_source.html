775 template <class dataTypeU, class dataTypeV>
776  inline int ReebSpace::computeGeometricalMeasures(Sheet3 &sheet){
777 
778  sheet.domainVolume_ = 0;
779  sheet.rangeArea_ = 0;
780  sheet.hyperVolume_ = 0;
781 
782  for(int i = 0; i < (int) sheet.tetList_.size(); i++){
783 
784  int tetId = sheet.tetList_[i];
785  vector<pair<double, double> > domainBox, rangeBox;
786  vector<vector<float> > domainPoints(4), rangePoints(4);
787 
788  for(int j = 0; j < 4; j++){
789  domainPoints[j].resize(3);
790  rangePoints[j].resize(2);
791 
792  int vertexId = -1;
793  triangulation_->getCellVertex(tetId, j, vertexId);
794 
795  triangulation_->getVertexPoint(vertexId,
796  domainPoints[j][0], domainPoints[j][1], domainPoints[j][2]);
797 
798  rangePoints[j][0] = ((dataTypeU *) uField_)[vertexId];
799  rangePoints[j][1] = ((dataTypeV *) vField_)[vertexId];
800  }
801 
802  Geometry::getBoundingBox(domainPoints, domainBox);
803  Geometry::getBoundingBox(rangePoints, rangeBox);
804 
805  sheet.domainVolume_ +=
806  (domainBox[0].second - domainBox[0].first)
807  *(domainBox[1].second - domainBox[1].first)
808  *(domainBox[2].second - domainBox[2].first);
809 
810  sheet.rangeArea_ +=
811  (rangeBox[0].second - rangeBox[0].first)
812  *(rangeBox[1].second - rangeBox[1].first);
813 
814  }
815 
816  if(sheet.domainVolume_){
817  sheet.hyperVolume_ = sheet.rangeArea_/sheet.domainVolume_;
818  }
819  else{
820  sheet.hyperVolume_ = 0;
821  }
822 
823  return 0;
824 }
