607 // *outputMinimum, vtkDataSet *outputJoinSaddle, vtkDataSet *outputSplitSaddle,
608 // vtkDataSet *outputMaximum, vtkUnstructuredGrid *outputJoinTree,
609 // vtkUnstructuredGrid *outputSplitTree){
610 
611  Memory m;
612 
613  vtkDataSet *input = inputs[0];
614  vtkDataSet *outputMinimum = outputs[0];
615  vtkDataSet *outputJoinSaddle = outputs[1];
616  vtkDataSet *outputSplitSaddle = outputs[2];
617  vtkDataSet *outputMaximum = outputs[3];
618  vtkUnstructuredGrid *outputJoinTree =
619  vtkUnstructuredGrid::SafeDownCast(outputs[4]);
620  vtkUnstructuredGrid *outputSplitTree =
621  vtkUnstructuredGrid::SafeDownCast(outputs[5]);
622 
623  // Check the last modification of the input
624  if(inputMTime_ != input->GetMTime()) {
625  inputMTime_ = input->GetMTime();
626  computeAll_ = true;
627  }
628 
629  // Use a pointer-base copy for the input data
630  outputMinimum->ShallowCopy(input);
631  outputJoinSaddle->ShallowCopy(input);
632  outputSplitSaddle->ShallowCopy(input);
633  outputMaximum->ShallowCopy(input);
634 
635  // Input data arrays
636  vtkDataArray *inputLowerBoundField = NULL;
637  vtkDataArray *inputUpperBoundField = NULL;
638 
639  // Get the upper bound field array in the input data set
640  if(upperBoundFiledName_.length()){
641  inputUpperBoundField =
642  input->GetPointData()->GetArray(upperBoundFiledName_.c_str());
643  } else {
644  inputUpperBoundField = input->GetPointData()->GetArray(upperBoundId);
645  }
646  // Error if not found
647  if(!inputUpperBoundField){
648  return -1;
649  }
650 
651  // Get the lower bound field array in the input data set
652  if(lowerBoundFieldName_.length()){
653  inputLowerBoundField =
654  input->GetPointData()->GetArray(lowerBoundFieldName_.c_str());
655  } else {
656  inputLowerBoundField = input->GetPointData()->GetArray(lowerBoundId);
657  }
658  // Error if not found
659  if(!inputLowerBoundField){
660  return -1;
661  }
662 
663  {
664  stringstream msg;
665  msg << "[vtkMandatoryCriticalPoints] Using `"
666  << inputLowerBoundField->GetName() << "' as lower bound..." << endl;
667  msg << "[vtkMandatoryCriticalPoints] Using `"
668  << inputUpperBoundField->GetName() << "' as upper bound..." << endl;
669  dMsg(cout, msg.str(), infoMsg);
670  }
671 
672  // Initialize the triangulation object with the input data set
673  triangulation_ = vtkTriangulation::getTriangulation(input);
674 
675  if(!triangulation_)
676  return -1;
677 
678  triangulation_->setWrapper(this);
679  mandatoryCriticalPoints_.setupTriangulation(triangulation_);
680 
681  bool hasChangedConnectivity = false;
682 
683  if((triangulation_->isEmpty())
684  ||(vtkTriangulation::hasChangedConnectivity(triangulation_, input, this))){
685  Modified();
686  hasChangedConnectivity = true;
687  }
688 
689  // Allocate the memory for the output scalar field
690  if((!outputMandatoryMinimum_)||(hasChangedConnectivity)){
691  if(!outputMandatoryMinimum_)
692  outputMandatoryMinimum_ = vtkIntArray::New();
693  outputMandatoryMinimum_->SetNumberOfTuples(input->GetNumberOfPoints());
694  outputMandatoryMinimum_->SetName("MinimumComponents");
695  }
696  if((!outputMandatoryJoinSaddle_)||(hasChangedConnectivity)){
697  if(!outputMandatoryJoinSaddle_)
698  outputMandatoryJoinSaddle_ = vtkIntArray::New();
699  outputMandatoryJoinSaddle_->SetNumberOfTuples(input->GetNumberOfPoints());
700  outputMandatoryJoinSaddle_->SetName("JoinSaddleComponents");
701  }
702  if((!outputMandatorySplitSaddle_)||(hasChangedConnectivity)){
703  if(!outputMandatorySplitSaddle_)
704  outputMandatorySplitSaddle_ = vtkIntArray::New();
705  outputMandatorySplitSaddle_->SetNumberOfTuples(input->GetNumberOfPoints());
706  outputMandatorySplitSaddle_->SetName("SplitSaddleComponents");
707  }
708  if((!outputMandatoryMaximum_)||(hasChangedConnectivity)){
709  if(!outputMandatoryMaximum_)
710  outputMandatoryMaximum_ = vtkIntArray::New();
711  outputMandatoryMaximum_->SetNumberOfTuples(input->GetNumberOfPoints());
712  outputMandatoryMaximum_->SetName("MaximumComponents");
713  }
714  // Add array in the output data set
715  outputMinimum->GetPointData()->AddArray(outputMandatoryMinimum_);
716  outputJoinSaddle->GetPointData()->AddArray(outputMandatoryJoinSaddle_);
717  outputSplitSaddle->GetPointData()->AddArray(outputMandatorySplitSaddle_);
718  outputMaximum->GetPointData()->AddArray(outputMandatoryMaximum_);
719 
720  // Calling the executing package
721  switch(inputUpperBoundField->GetDataType()){
722 
723  vtkTemplateMacro((
724  {
725  // Reset the baseCode object
726  if((computeAll_)||(hasChangedConnectivity))
727  mandatoryCriticalPoints_.flush();
728  // Wrapper
729  mandatoryCriticalPoints_.setWrapper(this);
730  // Set the number of vertex
731  mandatoryCriticalPoints_.setVertexNumber(input->GetNumberOfPoints());
732  // Set the coordinates of each vertex
733  double point[3];
734  for(int i=0 ; i<input->GetNumberOfPoints() ; i++){
735  input->GetPoint(i, point);
736  mandatoryCriticalPoints_.setVertexPosition(i,point);
737  }
738  // Set the void pointers to the upper and lower bound fields
739  mandatoryCriticalPoints_.setLowerBoundFieldPointer(
740  inputLowerBoundField->GetVoidPointer(0));
741  mandatoryCriticalPoints_.setUpperBoundFieldPointer(
742  inputUpperBoundField->GetVoidPointer(0));
743  // Set the output data pointers
744  mandatoryCriticalPoints_.setOutputMinimumDataPointer(
745  outputMandatoryMinimum_->GetVoidPointer(0));
746  mandatoryCriticalPoints_.setOutputJoinSaddleDataPointer(
747  outputMandatoryJoinSaddle_->GetVoidPointer(0));
748  mandatoryCriticalPoints_.setOutputSplitSaddleDataPointer(
749  outputMandatorySplitSaddle_->GetVoidPointer(0));
750  mandatoryCriticalPoints_.setOutputMaximumDataPointer(
751  outputMandatoryMaximum_->GetVoidPointer(0));
752  // Set the triangulation object
753  // Set the offsets
754  mandatoryCriticalPoints_.setSoSoffsets();
755  // Simplification threshold
756 
757  mandatoryCriticalPoints_.setSimplificationThreshold(
758  simplificationThreshold_);
759  // Execute process
760  if(computeAll_) {
761  mandatoryCriticalPoints_.execute<VTK_TT>();
762  computeAll_ = false;
763  simplify_ = false;
764  computeMinimumOutput_ = true;
765  computeJoinSaddleOutput_ = true;
766  computeSplitSaddleOutput_ = true;
767  computeMaximumOutput_ = true;
768  }
769  }
770  ));
771  }
772 
773  // Simplification
774  if(simplify_) {
775  // Simplify
776  mandatoryCriticalPoints_.simplifyJoinTree();
777  mandatoryCriticalPoints_.buildJoinTreePlanarLayout();
778  mandatoryCriticalPoints_.simplifySplitTree();
779  mandatoryCriticalPoints_.buildSplitTreePlanarLayout();
780  // Recompute the outputs
781  computeMinimumOutput_ = true;
782  computeJoinSaddleOutput_ = true;
783  computeSplitSaddleOutput_ = true;
784  computeMaximumOutput_ = true;
785  }
786  // Outputs
787  if(computeMinimumOutput_) {
788  if(outputAllMinimumComponents_)
789  mandatoryCriticalPoints_.outputAllMinima();
790  else
791  mandatoryCriticalPoints_.outputMinimum(outputMinimumComponentId_);
792  computeMinimumOutput_ = false;
793  }
794  if(computeJoinSaddleOutput_) {
795  if(outputAllJoinSaddleComponents_)
796  mandatoryCriticalPoints_.outputAllJoinSaddle();
797  else
798  mandatoryCriticalPoints_.outputJoinSaddle(outputJoinSaddleComponentId_);
799  computeJoinSaddleOutput_ = false;
800  }
801  if(computeSplitSaddleOutput_) {
802  if(outputAllSplitSaddleComponents_)
803  mandatoryCriticalPoints_.outputAllSplitSaddle();
804  else
805  mandatoryCriticalPoints_.outputSplitSaddle(outputSplitSaddleComponentId_);
806  computeSplitSaddleOutput_ = false;
807  }
808  if(computeMaximumOutput_) {
809  if(outputAllMaximumComponents_)
810  mandatoryCriticalPoints_.outputAllMaxima();
811  else
812  mandatoryCriticalPoints_.outputMaximum(outputMaximumComponentId_);
813  computeMaximumOutput_ = false;
814  }
815 
816  buildVtkTree(outputJoinTree, MandatoryCriticalPoints::TreeType::JoinTree);
817  buildVtkTree(outputSplitTree, MandatoryCriticalPoints::TreeType::SplitTree);
818 
819  {
820  stringstream msg;
821  memoryUsage_ += m.getElapsedUsage();
822  msg << "[vtkMandatoryCriticalPoints] Memory usage: " << memoryUsage_
823  << " MB." << endl;
824  dMsg(cout, msg.str(), memoryMsg);
825  }
826 
827  return 0;
828 }
