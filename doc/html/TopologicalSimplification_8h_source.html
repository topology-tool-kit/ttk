190  bool isMinima{true};

194  int neighbor;

248 template <typename dataType>
249 int TopologicalSimplification::addPerturbation(dataType* scalars, int* offsets) const{
250  dataType epsilon{};
251 
252  if(is_same<dataType,double>::value) epsilon=pow10(1-DBL_DIG);
253  else if(is_same<dataType,float>::value) epsilon=pow10(1-FLT_DIG);
254  else return -1;
255 
256  vector<tuple<dataType,int,int>> perturbation(vertexNumber_);
257  for(int i=0; i<vertexNumber_; ++i){
258  get<0>(perturbation[i])=scalars[i];
259  get<1>(perturbation[i])=offsets[i];
260  get<2>(perturbation[i])=i;
261  }
262 
263  SweepCmp cmp(true);
264  sort(perturbation.begin(), perturbation.end(), cmp);
265 
266  for(int i=0; i<vertexNumber_; ++i){
267  if(i){
268  if(get<0>(perturbation[i]) <= get<0>(perturbation[i-1]))
269  get<0>(perturbation[i])=get<0>(perturbation[i-1]) + epsilon;
270  }
271  scalars[get<2>(perturbation[i])]=get<0>(perturbation[i]);
272  }
273 
274  return 0;
275 }

278 int TopologicalSimplification::execute() const{
279  // get input data
280  dataType* inputScalars=static_cast<dataType*>(inputScalarFieldPointer_);
281  dataType* scalars=static_cast<dataType*>(outputScalarFieldPointer_);
282  int* identifiers=static_cast<int*>(vertexIdentifierScalarFieldPointer_);
283  int* inputOffsets=static_cast<int*>(inputOffsetScalarFieldPointer_);
284  int* offsets=static_cast<int*>(outputOffsetScalarFieldPointer_);
285 
286  Timer t;
287 
288  // pre-processing
289  for(int k=0; k<vertexNumber_; ++k){
290  scalars[k]=inputScalars[k];
291  offsets[k]=inputOffsets[k];
292  }
293 
294  // get the user extremum list
295  vector<bool> extrema(vertexNumber_, false);
296  for(int k=0; k<constraintNumber_; ++k) extrema[identifiers[k]]=true;
297 
298  vector<int> authorizedMinima;
299  vector<int> authorizedMaxima;
300  vector<bool> authorizedExtrema(vertexNumber_, false);
301  getCriticalPoints<dataType>(scalars,offsets,authorizedMinima,authorizedMaxima,extrema);
302 
303  // declare the tuple-comparison functor
304  SweepCmp cmp;
305 
306  // processing
307  int iteration{};
308  for(int i=0; i<vertexNumber_; ++i){
309  for(int j=0; j<2; ++j){
310  bool isIncreasingOrder=!j;
311 
312  cmp.setIsIncreasingOrder(isIncreasingOrder);
313  set<tuple<dataType,int,int>, decltype(cmp)> sweepFront(cmp);
314  vector<bool> visitedVertices(vertexNumber_, false);
315  vector<int> adjustmentSequence(vertexNumber_);
316 
317  // add the seeds
318  if(isIncreasingOrder){
319  for(int k : authorizedMinima){
320  authorizedExtrema[k]=true;
321  sweepFront.emplace(scalars[k],offsets[k],k);
322  visitedVertices[k]=true;
323  }
324  }
325  else{
326  for(int k : authorizedMaxima){
327  authorizedExtrema[k]=true;
328  sweepFront.emplace(scalars[k],offsets[k],k);
329  visitedVertices[k]=true;
330  }
331  }
332 
333  // growth by neighborhood of the seeds
334  int adjustmentPos{};
335  do{
336  auto front=sweepFront.begin();
337  if(front==sweepFront.end()) return -1;
338 
339  int vertexId=get<2>(*front);
340  sweepFront.erase(front);
341 
342  int neighborNumber=triangulation_->getVertexNeighborNumber(vertexId);
343  for(int k=0; k<neighborNumber; ++k){
344  int neighbor;
345  triangulation_->getVertexNeighbor(vertexId,k,neighbor);
346  if(!visitedVertices[neighbor]){
347  sweepFront.emplace(scalars[neighbor],offsets[neighbor],neighbor);
348  visitedVertices[neighbor]=true;
349  }
350  }
351  adjustmentSequence[adjustmentPos]=vertexId;
352  ++adjustmentPos;
353  }while(!sweepFront.empty());
354 
355  // save offsets and rearrange scalars
356  int offset=(isIncreasingOrder)?0:vertexNumber_+1;
357  for(int k=0; k<vertexNumber_; ++k){
358  if(isIncreasingOrder){
359  if(k and scalars[adjustmentSequence[k]] <= scalars[adjustmentSequence[k-1]])
360  scalars[adjustmentSequence[k]]=scalars[adjustmentSequence[k-1]];
361  ++offset;
362  }
363  else{
364  if(k and scalars[adjustmentSequence[k]] >= scalars[adjustmentSequence[k-1]])
365  scalars[adjustmentSequence[k]]=scalars[adjustmentSequence[k-1]];
366  --offset;
367  }
368  offsets[adjustmentSequence[k]]=offset;
369  }
370  }
371 
372  // test convergence
373  bool needForMoreIterations{false};
374  vector<int> minima;
375  vector<int> maxima;
376  getCriticalPoints<dataType>(scalars,offsets,minima,maxima);
377 
378  if(maxima.size() > authorizedMaxima.size()) needForMoreIterations=true;
379  if(minima.size() > authorizedMinima.size()) needForMoreIterations=true;
380  if(!needForMoreIterations){
381  for(int k : minima){
382  if(!authorizedExtrema[k]){
383  needForMoreIterations=true;
384  break;
385  }
386  }
387  }
388  if(!needForMoreIterations){
389  for(int k : maxima){
390  if(!authorizedExtrema[k]){
391  needForMoreIterations=true;
392  break;
393  }
394  }
395  }
396 
397  // optional adding of perturbation
398  if(addPerturbation_) addPerturbation<dataType>(scalars, offsets);
399 
400  ++iteration;
401  if(!needForMoreIterations) break;
402  }
403

407  << " in " << t.getElapsedTime() << " s. (" << threadNumber_
408  << " threads(s), " << iteration << " ite.)."
409  << endl;

411  }
412 
413  return 0;
