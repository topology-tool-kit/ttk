104 int PersistenceCurve::computePersistencePlot(const vector<tuple<idVertex, idVertex, scalarType>>& pairs,
105  vector<pair<scalarType, idVertex>> &plot) const{
106 
107  idVertex nbElmnt = pairs.size();
108  plot.resize(nbElmnt);
109 
110  // build curve
111  const scalarType epsilon=static_cast<scalarType>(pow(10, -REAL_SIGNIFICANT_DIGITS));
112  for (idVertex i = 0; i < nbElmnt; ++i) {
113  plot[i].first = std::max(get<2>(pairs[i]), epsilon);
